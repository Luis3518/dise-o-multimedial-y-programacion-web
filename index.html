Codigo:


<script src="https://ajax.googleapis.com/ajax/libs/jquery/3.3.1/jquery.min.js">
</script>
<script>
$(document).ready(function(){
$("#hide").click(function(){
$(".test").hide();
});
$("#show").click(function(){
$(".test").show();
});
});
</script>
<button id="hide">Ocultar</button>
<button id="show">Mostrar</button>
<h2 class="test">Titulo del texto a ocultar</h2>
